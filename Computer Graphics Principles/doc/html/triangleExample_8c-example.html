triangleExample.c
The figure shows the output of the triangle example.

If rendering pipeline works, this example renders one colored triangle.

Initialization

All global variables are stored in one place - in a structure.

struct TriangleExampleVariables{
GPU gpu;
ProgramID program;
VertexPullerID puller;
}triangleExample;

View and projection matrices are stored in extern variables:

extern Mat4 projectionMatrix;
extern Mat4 viewMatrix;

When a mouse is moving, these matrices are recomputed.
When example is started, function triangleExample_onInit() is called. This function creates gpu, initializes viewport and initializes matrices:

triangleExample.gpu = cpu_createGPU();
cpu_setViewportSize(triangleExample.gpu,(size_t)width,(size_t)height);
cpu_initMatrices(width,height);

Then, the function reserves uniforms on gpu:

cpu_reserveUniform(triangleExample.gpu,"projectionMatrix",UNIFORM_MAT4);
cpu_reserveUniform(triangleExample.gpu,"viewMatrix" ,UNIFORM_MAT4);

Next, the function creates shader program

triangleExample.program = cpu_createProgram(triangleExample.gpu);

Shaders has to be attach to program:

cpu_attachVertexShader (triangleExample.gpu,triangleExample.program,triangleExample_vertexShader );
cpu_attachFragmentShader(triangleExample.gpu,triangleExample.program,triangleExample_fragmentShader);

One vertex attribute (color) is interpolated during rasterization, we have to set its type, location and type of interpolation:

cpu_setAttributeInterpolation(triangleExample.gpu,triangleExample.program,0,ATTRIB_VEC3,SMOOTH);

Last thing that needs to be done in initialization is creation of empty vertex puller:

cpu_createVertexPullers(triangleExample.gpu,1,&triangleExample.puller);

Deinitialization

At the end of application, all allocated resource has to be released in function triangleExample_onExit:

cpu_destroyGPU(triangleExample.gpu);

Drawing

Function triangleExample_onDraw is called every frame. This function clears depth buffer and color buffer:

cpu_clearDepth(triangleExample.gpu,(float)(+INFINITY));
Vec4 color;
init_Vec4(&color,.1f,.1f,.1f,1.f);
cpu_clearColor(triangleExample.gpu,&color);

Function then activates program:

cpu_useProgram(triangleExample.gpu,triangleExample.program);

Then, it activates vertex puller:

cpu_bindVertexPuller(triangleExample.gpu,triangleExample.puller);

Next, it has to upload uniforms to gpu:

cpu_uniformMatrix4fv(triangleExample.gpu,getUniformLocation(triangleExample.gpu,"viewMatrix" ),(float*)&viewMatrix );
cpu_uniformMatrix4fv(triangleExample.gpu,getUniformLocation(triangleExample.gpu,"projectionMatrix"),(float*)&projectionMatrix);

Now, it is possible to execute draw call:

cpu_drawTriangles(triangleExample.gpu,3);

Lastly, the function has to swap framebuffer into window surface:

cpu_swapBuffers(surface,triangleExample.gpu);

Shaders

Vertex shader starts with creation of pointers to uniform matrices:

Mat4 const*const view = shader_interpretUniformAsMat4(gpu_getUniformsHandle(gpu),getUniformLocation(gpu,"viewMatrix" ));
Mat4 const*const proj = shader_interpretUniformAsMat4(gpu_getUniformsHandle(gpu),getUniformLocation(gpu,"projectionMatrix"));

Then, it projects generated vertex position into clip-space:

Mat4 mvp;
multiply_Mat4_Mat4(&mvp,proj,view);
Vec4 position;
init_Vec4(
&position,
-1.f+2.f*(float)(input->gl_VertexID%2),
-1.f+2.f*(float)(input->gl_VertexID/2),
0.f,
1.f);
multiply_Mat4_Vec4(
&output->gl_Position,
&mvp,
&position);

Then, it gets pointer handle to zeroth output vertex attribute:

Vec3 *const colorAttribute = vs_interpretOutputVertexAttributeAsVec3(gpu,output,0);

Finaly, it writes generated color into this output vertex attribute:

init_Vec3(
colorAttribute,
(float)(input->gl_VertexID == 0),
(float)(input->gl_VertexID == 1),
(float)(input->gl_VertexID == 2));

Fragment shader starts with creation of pointers to zeroth input fragment attribute:

Vec3 const*colorAttribute = fs_interpretInputAttributeAsVec3(gpu,input,0);

This attribute contains interpolated color. At the end, fragment shader writes color to output fragment.

copy_Vec3Float_To_Vec4(
&output->color,
colorAttribute,
1.f);

void triangleExample_vertexShader(
GPUVertexShaderOutput *const output,
GPUVertexShaderInput const*const input ,
GPU const gpu ){
assert(output != NULL);
assert(input != NULL);
assert(gpu != NULL);
Mat4 const*const view = shader_interpretUniformAsMat4(gpu_getUniformsHandle(gpu),getUniformLocation(gpu,"viewMatrix" ));
Mat4 const*const proj = shader_interpretUniformAsMat4(gpu_getUniformsHandle(gpu),getUniformLocation(gpu,"projectionMatrix"));
Mat4 mvp;
multiply_Mat4_Mat4(&mvp,proj,view);
Vec4 position;
init_Vec4(
&position,
-1.f+2.f*(float)(input->gl_VertexID%2),
-1.f+2.f*(float)(input->gl_VertexID/2),
0.f,
1.f);
multiply_Mat4_Vec4(
&output->gl_Position,
&mvp,
&position);
Vec3 *const colorAttribute = vs_interpretOutputVertexAttributeAsVec3(gpu,output,0);
init_Vec3(
colorAttribute,
(float)(input->gl_VertexID == 0),
(float)(input->gl_VertexID == 1),
(float)(input->gl_VertexID == 2));
}
void triangleExample_fragmentShader(
GPUFragmentShaderOutput *const output,
GPUFragmentShaderInput const*const input ,
GPU const gpu ){
assert(output != NULL);
assert(input != NULL);
assert(gpu != NULL);
Vec3 const*colorAttribute = fs_interpretInputAttributeAsVec3(gpu,input,0);
copy_Vec3Float_To_Vec4(
&output->color,
colorAttribute,
1.f);
}
void triangleExample_onInit(int32_t width,int32_t height){
triangleExample.gpu = cpu_createGPU();
cpu_setViewportSize(triangleExample.gpu,(size_t)width,(size_t)height);
cpu_initMatrices(width,height);
cpu_reserveUniform(triangleExample.gpu,"projectionMatrix",UNIFORM_MAT4);
cpu_reserveUniform(triangleExample.gpu,"viewMatrix" ,UNIFORM_MAT4);
triangleExample.program = cpu_createProgram(triangleExample.gpu);
cpu_attachVertexShader (triangleExample.gpu,triangleExample.program,triangleExample_vertexShader );
cpu_attachFragmentShader(triangleExample.gpu,triangleExample.program,triangleExample_fragmentShader);
cpu_setAttributeInterpolation(triangleExample.gpu,triangleExample.program,0,ATTRIB_VEC3,SMOOTH);
cpu_createVertexPullers(triangleExample.gpu,1,&triangleExample.puller);
}
void triangleExample_onExit(){
cpu_destroyGPU(triangleExample.gpu);
}
void triangleExample_onDraw(SDL_Surface*surface){
assert(surface != NULL);
cpu_clearDepth(triangleExample.gpu,(float)(+INFINITY));
Vec4 color;
init_Vec4(&color,.1f,.1f,.1f,1.f);
cpu_clearColor(triangleExample.gpu,&color);
cpu_useProgram(triangleExample.gpu,triangleExample.program);
cpu_bindVertexPuller(triangleExample.gpu,triangleExample.puller);
cpu_uniformMatrix4fv(triangleExample.gpu,getUniformLocation(triangleExample.gpu,"viewMatrix" ),(float*)&viewMatrix );
cpu_uniformMatrix4fv(triangleExample.gpu,getUniformLocation(triangleExample.gpu,"projectionMatrix"),(float*)&projectionMatrix);
cpu_drawTriangles(triangleExample.gpu,3);
cpu_swapBuffers(surface,triangleExample.gpu);
}
